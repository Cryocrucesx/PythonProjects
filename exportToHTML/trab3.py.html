def problema1():
    valores = []
    while True:
        valor = float(input("Digite um valor real (ou -1 para sair): "))
        if valor == -1:
            break
        valores.append(valor)

    quantidade = len(valores)
    soma = sum(valores)
    media = soma / quantidade if quantidade > 0 else 0
    maiores_que_20 = sum(1 for v in valores if v > 20)

    print(f"Quantidade de valores digitados: {quantidade}")
    print(f"Soma dos valores: {soma}")
    print(f"Média aritmética: {media}")
    print(f"Quantidade de valores maior que 20: {maiores_que_20}")


def problema2():
    aprovados = 0
    reprovados = 0
    total_alunos = 0

    while True:
        nota = float(input("Digite a nota do aluno (ou -1 para sair): "))
        if nota == -1:
            break
        total_alunos += 1
        if nota >= 5:
            aprovados += 1
        else:
            reprovados += 1

    print(f"Total de alunos que fizeram a prova: {total_alunos}")
    print(f"Quantidade de alunos aprovados: {aprovados}")
    print(f"Quantidade de alunos reprovados: {reprovados}")


def problema3():
    soma_pares = 0
    quantidade_pares = 0
    soma_impares = 0
    quantidade_impares = 0

    while True:
        numero = int(input("Digite um número inteiro (ou 0 para sair): "))
        if numero == 0:
            break
        if numero % 2 == 0:
            soma_pares += numero
            quantidade_pares += 1
        else:
            soma_impares += numero
            quantidade_impares += 1

    media_pares = soma_pares / quantidade_pares if quantidade_pares > 0 else 0
    media_impares = soma_impares / quantidade_impares if quantidade_impares > 0 else 0

    print(f"Média aritmética dos números pares: {media_pares}")
    print(f"Média aritmética dos números ímpares: {media_impares}")


# Chamando as funções para execução
tipo_problema = int(input("Escolha o problema para executar (1, 2 ou 3): "))
if tipo_problema == 1:
    problema1()
elif tipo_problema == 2:
    problema2()
elif tipo_problema == 3:
    problema3()
else:
    print("Opção inválida!")
